import { grade4Data } from "../data/grade4.js"; import { appState } from "../state/appState.js"; import { goToMission, goToWorldMap } from "../utils/navigation.js"; import { isMissionCompleted } from "../utils/progress.js"; export function renderSubmap() { const world = grade4Data.worlds.find(w => w.id === appState.selectedWorldId); const appShell = document.querySelector(".app-shell"); if (!world || !appShell) return; appShell.innerHTML = `
Nivel 4
${world.title}
${appState.coins}
${appState.gems}
Submapa

${world.title}

${world.description}

${world.missions.map((mission, index) => { const completed = isMissionCompleted(mission.id); const available = completed || mission.status === "current" || mission.status === "completed"; const statusText = completed ? "Completada" : available ? "Disponible" : "Bloqueada"; return `
${index + 1}

${mission.title}

Dificultad: ${mission.difficulty}

Estado: ${statusText}

`; }).join("")}
`; document.getElementById("backToWorldMapBtn")?.addEventListener("click", () => { goToWorldMap(); window.renderApp(); }); document.querySelectorAll(".openMissionBtn").forEach(btn => { btn.addEventListener("click", () => { goToMission(world.id, btn.dataset.missionId); window.renderApp(); }); }); }
